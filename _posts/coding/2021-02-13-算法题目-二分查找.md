class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        i = m-1
        j = n-1
        while i>=0 and j>=0:
            if nums1[i] < nums2[j]:
                nums1[i+j+1] = nums2[j]
                j -= 1
            else:

class Solution:
    def search(self, nums, target: int) -> int:
        if not nums:
            return -1
        l, r = 0, len(nums) - 1
        while l <= r:
            mid = (l + r)//2
            if nums[mid] == target: return mid
            if nums[mid] > nums[0]:  # è¾¹ç•Œæ¡ä»¶æ˜¯æœ€å·¦æœ€å³
                if nums[0] <= target < nums[mid]:  # æ³¨æ„æœ‰ç­‰äº
                    r = mid - 1
                else:
                    l = mid + 1
            else:
                if nums[len(nums)-1] >= target > nums[mid] :
                    l = mid + 1
                else: 
                    r = mid - 1
        return  -1
